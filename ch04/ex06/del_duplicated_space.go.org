package main

import "fmt"
import "unicode"

func del_duplicated(bytes []byte) []byte {
	j := 0
	var tmp byte
	for i := 0; i <len(bytes); i++ {
		if !unicode.IsSpace(tmp) {
			bytes[j] = bytes[i]
			j++
		}
		tmp = bytes[i]
	}
	return bytes[:j]
}
